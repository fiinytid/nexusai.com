const PROXIES = [
  url => `https://corsproxy.io/?${encodeURIComponent(url)}`,
  url => `https://api.allorigins.win/raw?url=${encodeURIComponent(url)}`,
];
async function fetchRoblox(url) {
  for (const mkP of PROXIES) {
    try {
      const r = await fetch(mkP(url), { signal:AbortSignal.timeout(8000) });
      if (!r.ok) continue;
      const txt = await r.text();
      return JSON.parse(txt);
    } catch(e) {}
  }
  // Try direct
  try { const r = await fetch(url); if (r.ok) return r.json(); } catch(e) {}
  throw new Error('Cannot reach Roblox API. Check internet connection.');
}

async function findUser() {

  document.getElementById('findBtn').disabled = true;
  try {
    const d = await fetchRoblox(`https://users.roblox.com/v1/users/search?keyword=${encodeURIComponent(un)}&limit=10`);
    if (!d.data?.length) throw new Error(`Account "${un}" not found on Roblox`);
    const u = d.data.find(x=>x.name.toLowerCase()===un.toLowerCase()) || d.data[0];
    document.getElementById('authLoadTxt').textContent = 'Loading avatar...';
    // FIX: correct Roblox thumbnail API URL format
    let avUrl = `https://www.roblox.com/headshot-thumbnail/image?userId=${u.id}&width=420&height=420&format=png`;
    try {
      const td = await fetchRoblox(`https://thumbnails.roblox.com/v1/users/avatar-headshot?userIds=${u.id}&size=420x420&format=Png`);
      if (td.data?.[0]?.imageUrl) avUrl = td.data[0].imageUrl;
    } catch(e) {}
    pendUser = { id:u.id, username:u.name, displayName:u.displayName||u.name, avatar:avUrl };
    document.getElementById('authAv').src = avUrl;
    document.getElementById('authName').textContent = u.name;
    document.getElementById('authId').textContent = 'ID: '+u.id;
    document.getElementById('authNameBtn').textContent = u.name;
    document.getElementById('authPrev').classList.add('show');
  } catch(e) { showAErr('authErr', e.message); }
  finally {
    document.getElementById('authLoad').classList.remove('show');
    document.getElementById('findBtn').disabled = false;
  }
}

async function send() {
  if(S.gen) return;
  const txt=document.getElementById('inp').value.trim();
  const attachments=[...S.attachments];
  if(!txt&&!attachments.length) return;

  if(!isOwner()&&S.credits<=-5){toast('❌ Credits depleted! Wait for daily credits or buy more.');openShop();return;}
  if(!isOwner()&&S.model.cost>0&&S.creditsm.role==='user').length===1;
    const greetCtx = `\n\nUser Roblox username: "${S.user?.username||'Unknown'}". Plan: ${S.plan||'free'}. ${isFirst?'First message - greet by username!':'Continue conversation.'}`;
    const sysWithGreet = SYS + greetCtx;

    if(sendModel.provider==='gemini')          r=await callGemini(sendModel.id, attachments, sig, sysWithGreet);
    else if(sendModel.provider==='grok')       r=await callGrok(sendModel.id, attachments, sig, sysWithGreet);
    else if(sendModel.provider==='claude')     r=await callClaude(sendModel.id, attachments, sig, sysWithGreet);
    else if(sendModel.provider==='groqcloud')  r=await callGroqCloud(sendModel.id, attachments, sig, sysWithGreet);
    else                                       r=await callOpenAI(sendModel.id, attachments, sig, sysWithGreet);

    rmThinking();
    const am={role:'ai',content:r,model:sendModel.label,provider:sendModel.provider,time:Date.now()};

    // Push to the CORRECT conversation (sendConvId), even if user switched conv
    const sendConv = S.convs.find(x=>x.id===sendConvId);
    if(sendConv) {
      sendConv.msgs.push(am);
      // Only DOM-update if user is still looking at this conversation
      if(S.curConv===sendConvId) {
        S.msgs = sendConv.msgs;
        appendMsg(am);
        const lb=document.querySelector('#msgs .msg.ai:last-child .bubble');
        if(lb) showTypingAnim(lb);
      } else {
        // User switched conv — update title silently
        toast('💬 Response ready in: '+c.title);
      }
    }

    // Credit deduction (decimal, based on complexity)
    if(!isOwner()&&sendModel.cost>0){
      const complexity=txt.length>300?1.0:txt.length>80?0.6:0.2;
      S.credits=parseFloat(Math.max(-5,S.credits-sendModel.cost*complexity).toFixed(2));
      updateCreds();
    } else if(!isOwner()&&sendModel.cost===0){
      const isCode=/gui|script|code|build|create|function|system|class/i.test(txt);
      if(isCode) S.credits=parseFloat(Math.max(-5,S.credits-0.01).toFixed(2));
      updateCreds();
    }

    saveS();addLog('info',sendModel.label+' responded');
    await setConvTitle(sendConvId);
    renderConvs();
  } catch(e) {
    rmThinking();
    if(e.name==='AbortError') return;
    const em=e.message||'Unknown error';
    let fe=`❌ **Response failed**\n\n${em}\n\nTry switching to a different model.`;
    if(em.toLowerCase().includes('network')||em.includes('fetch')||em.includes('Failed to fetch')){
      fe=`❌ **Connection failed**\n\nCould not reach the AI server. Check your internet connection.`;
    }
    const errMsg={role:'ai',content:fe,model:S.model.label,provider:S.model.provider,time:Date.now()};
    // Save error to correct conv
    const sendConv=S.convs.find(x=>x.id===sendConvId);
    if(sendConv) sendConv.msgs.push(errMsg);
    if(S.curConv===sendConvId) appendMsg(errMsg);
    addLog('err',em);
    toast('❌ Response failed — try another model');
  } finally {
    S.gen=false;S.cancelCtrl=null;

    document.getElementById('sendBtn').classList.remove('hidden');
    document.getElementById('cancelBtn').classList.add('hidden');
    document.getElementById('sendBtn').disabled=false;
  }
}

function buildMsgs() {
  return S.msgs.slice(-24).map(m=>{
    let content = m.content||'';
    // Include file content in messages
    const files = (m.attachments||[]).filter(a=>a.type==='file'&&a.content);
    if(files.length) {
      content += '\n\n' + files.map(f=>`[File: ${f.name}]\n\`\`\`\n${f.content.substring(0,8000)}\n\`\`\``).join('\n\n');
    }
    return {role:m.role==='ai'?'assistant':'user', content};
  });
}

async function callGemini(model, attachments, signal, sys) {
  const ms=buildMsgs();
  const contents=ms.map(m=>({role:m.role==='assistant'?'model':'user',parts:[{text:m.content}]}));
  // Add image attachments to last user message
  const imgs=attachments.filter(a=>a.type==='image');
  if(imgs.length&&contents.length>0){
    const l=contents[contents.length-1];
    if(l.role==='user') imgs.forEach(a=>l.parts.push({inline_data:{mime_type:a.mime||'image/jpeg',data:a.dataUrl.split(',')[1]}}));
  }
  const r=await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${K.gemini}`,{
    method:'POST',signal,headers:{'Content-Type':'application/json'},
    body:JSON.stringify({contents,systemInstruction:{parts:[{text:sys||SYS}]},generationConfig:{maxOutputTokens:16384,temperature:.72}})
  });
  if(!r.ok){const e=await r.json().catch(()=>({}));throw new Error(e.error?.message||`Gemini ${r.status}`);}
  const d=await r.json();
  if(d.candidates?.[0]?.content?.parts?.[0]?.text) return d.candidates[0].content.parts[0].text;
  throw new Error('Empty Gemini response');
}
